#library "mainacs"
#include "zcommon.acs"


int lv2 = 100;
int lv3 = 200;
int lv4 = 300;
int lv5 = 400;


script "levelup" (void)
{
	SetFont("BFONT");
	HudMessage(s:"LEVEL UP ^"; HUDMSG_FADEOUT, 1, CR_WHITE, 0.5, 0.5, 270, 1.0);
}
script "leveldown" (void)
{
	SetFont("BFONT");
	HudMessage(s:"LEVEL DOWN |"; HUDMSG_FADEOUT, 1, CR_WHITE, 0.5, 0.5, 270, 1.0);
}

script "getarmor" (void)
{
	SetResultValue(GetArmorInfo(ARMORINFO_SAVEAMOUNT));
}

script "chklvl" (void)
{
	int lvl = CheckInventory("lvl");
	int scrap = CheckInventory("Scrap");
	
	if (lvl == 1 && scrap >= lv2) { GiveInventory("lvl", 1); ACS_NamedExecute("levelup", 0); }
	else
	if (lvl == 2 && scrap >= lv3) { GiveInventory("lvl", 1); ACS_NamedExecute("levelup", 0); }
	else
	if (lvl == 3 && scrap >= lv4) { GiveInventory("lvl", 1); ACS_NamedExecute("levelup", 0); }
	else
	if (lvl == 4 && scrap >= lv5) { GiveInventory("lvl", 1); ACS_NamedExecute("levelup", 0); }
	
	if (lvl == 2 && scrap <= lv2 - 1) { TakeInventory("lvl", 1); ACS_NamedExecute("leveldown", 0); }
	else
	if (lvl == 3 && scrap <= lv3 - 1) { TakeInventory("lvl", 1); ACS_NamedExecute("leveldown", 0); }
	else
	if (lvl == 4 && scrap <= lv4 - 1) { TakeInventory("lvl", 1); ACS_NamedExecute("leveldown", 0); }
	else
	if (lvl == 5 && scrap <= lv5 - 1) { TakeInventory("lvl", 1); ACS_NamedExecute("leveldown", 0); }
}

script "resetalt" ENTER
{
	TakeInventory("paltcooldown", 999);
}
