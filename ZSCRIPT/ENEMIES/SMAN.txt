	}

  {
  DropItem "SAmo";
  Decal "bhole";
  }
  
	States
	{
		Missile:
			SPOS E 10 A_FaceTarget();
			SPOS F 0 bright A_PlaySound("shotguy/attack", CHAN_WEAPON);

			SPOS E 10;
			Goto See;
	}
}
