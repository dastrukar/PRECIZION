class SM : Actor replaces ShotgunGuy
{
	States
	{
		Spawn:
			TNT1 A 0;
			TNT1 A 0 A_Jump(256,"S1","S2");
		S1:
			TNT1 A 0 A_SpawnItemEx("SSGMan");
			stop;
		S2:
			TNT1 A 0 A_SpawnItemEx("SMan");
			stop;
	}
}

class SMan : ShotgunGuy
{
Default
  {
  DropItem "SAmo";
  }
  
	States
	{
		Missile:
			SPOS E 10 A_FaceTarget();
			SPOS F 0 bright A_PlaySound("shotguy/attack", CHAN_WEAPON);
			SPOS F 10 bright A_CustomBulletAttack (22.5, 0, 3, 0, "BulletPuff", 0, CBAF_NORANDOM, AAPTR_TARGET, "zmanblt");
			SPOS E 10;
			Goto See;
		Death:
			TNT1 A 0 ACS_NamedExecuteWithResult("scoreint", s_sman);
			SPOS H 5;
			SPOS I 5 A_Scream();
			SPOS J 5 A_NoBlocking();
			SPOS K 5;
			SPOS L -1;
			Stop;
		XDeath:
			TNT1 A 0 ACS_NamedExecuteWithResult("scoreint", s_sman);
			SPOS M 5;
			SPOS N 5 A_XScream();
			SPOS O 5 A_NoBlocking();
			SPOS PQRST 5;
			SPOS U -1;
			Stop;
	}
}
