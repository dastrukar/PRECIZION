//used for weapon pickups

class Handgun : GItem replaces Pistol
{
Default
  {
  Inventory.PickupMessage "Picked up the Handgun.";
  }
  
	States
	{
		Spawn:
			HNDG A -1;
			stop;
			
		Pickup:
			TNT1 A 0 A_GiveGun("pp","Handgun");
			stop;
	}
}
