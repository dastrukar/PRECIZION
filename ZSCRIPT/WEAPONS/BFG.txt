//it's called the BFG9000 for a reason..
class BFG9k : Gun replaces BFG9000
{
Default
  {
  Inventory.PickupMessage "You got the BFG9000, fuck yeah.";
  Weapon.AmmoType1 "bfgrounds";
  Weapon.AmmoType2 "bfgrounds";
  Weapon.AmmoUse1 10;
  Weapon.AmmoUse2 1;
  Weapon.AmmoGive 100;
  }
  
	States
	{
		Ready:
			BFGG A 1 A_WeaponReady();
			Loop;
		Deselect:
			BFGG A 1 A_Lower();
			Loop;
		Select:
			BFGG A 1 A_Raise();
			Loop;
		Fire:
			//BFGG A 20 A_BFGSound();
			BFGG A 0 A_TakeInventory("bfgrounds", invoker.AmmoUse1);
			BFGG B 5 A_GunFlash();
			BFGG BBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBB 0 A_FireProjectile("bfgballs", frandom(-11, 11), false, 0, 0, 0, frandom(-11, 11));
			BFGG B 5;
			BFGG B 10;
			Goto Ready;
		AltFire:
			//BFGG B 5 A_GunFlash();
			BFGG A 0 A_TakeInventory("bfgrounds", invoker.AmmoUse2);
			BFGG BBBBBBBBBB 1 A_FireProjectile("bfgballs", frandom(-11, 11), false, 0, 0, 0, frandom(-11, 11));
			BFGG B 5 A_ReFire();
			BFGG B 10;
			Goto Ready;
			
		Flash:
			BFGF A 2 Bright A_Light1();
			BFGF B 3 Bright A_Light2();
			Goto LightDone;
		Spawn:
			BFUG A -1;
			Stop;
  }
}
