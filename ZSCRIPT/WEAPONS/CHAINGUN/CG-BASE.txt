class cg : Gun
{
Default
  {
  Weapon.AmmoUse 1;
  Weapon.AmmoGive 50;
  Weapon.AmmoType "PBullet";
  Obituary "%o got mowed down by %k's Chaingun.";
  Decal "bhole";
  Gun.Pickup "PChain";
  Gun.Throw "flycg";
  Gun.Name "cg";
  }
  
	States
	{
		Ready:
			CHGG A 1 A_WeaponReady();
			loop;
			
		Deselect:
			CHGG A 1 A_Lower(10);
			loop;
			
		Select:
			MISG A 0 Offset(-20, 50);
		Select1:
			CHGG A 1 A_Raise(10);
			loop;
			
		Flash:
			CHGF A 3 Bright A_Light(5);
			CHGF B 3 Bright A_Light(5);
			goto LightDone;
   }
}
